const express = require('express');
const axios = require('axios');

const app = express();
const PORT = 3000;

// Middleware para que acepte JSON en el body
app.use(express.json());

// ğŸš¨ CONFIGURA ESTO
const NOTION_TOKEN = 'TU_TOKEN_SECRETO_DE_NOTION';
const DATABASE_ID = 'TU_ID_DE_BASE_DE_DATOS';

// ğŸ”§ Utilidad para obtener la fecha de hace 14 dÃ­as
function getCutoffDate() {
  const date = new Date();
  date.setDate(date.getDate() - 14);
  return date.toISOString();
}

// ğŸ“© Ruta para recibir tareas desde tu formulario
app.post('/webhook', async (req, res) => {
  const data = req.body;
  console.log('ğŸ“¥ Recibido desde el formulario:', data);

  try {
    // Enviar los datos a Notion
    const response = await axios.post(
      'https://api.notion.com/v1/pages',
      {
        parent: { database_id: DATABASE_ID },
        properties: {
          Tarea: {
            title: data.Tarea
          },
          Estado: {
            status: data.Estado
          },
          Prioridad: {
            select: data.Prioridad
          },
          Ãrea: {
            select: data.Ãrea
          },
          'Fecha de vencimiento': {
            date: data['Fecha de vencimiento']
          }
        }
      },
      {
        headers: {
          Authorization: `Bearer ${NOTION_TOKEN}`,
          'Notion-Version': '2022-06-28',
          'Content-Type': 'application/json'
        }
      }
    );

    res.send('âœ… Tarea recibida y guardada en Notion.');
  } catch (error) {
    console.error('âŒ Error al enviar a Notion:', error.response?.data || error.message);
    res.status(500).send('âš ï¸ Error al enviar tarea a Notion.');
  }
});

// ğŸ§¹ Ruta para eliminar tareas con status "Done" hace +14 dÃ­as
app.get('/cleanup', async (req, res) => {
  const cutoffDate = getCutoffDate();

  try {
    const response = await axios.post(
      `https://api.notion.com/v1/databases/${DATABASE_ID}/query`,
      {
        filter: {
          and: [
            {
              property: 'Status',
              select: { equals: 'Done' }
            },
            {
              property: 'Fecha de completado',
              date: { before: cutoffDate }
            }
          ]
        }
      },
      {
        headers: {
          Authorization: `Bearer ${NOTION_TOKEN}`,
          'Notion-Version': '2022-06-28',
          'Content-Type': 'application/json'
        }
      }
    );

    const pages = response.data.results;
    console.log(`ğŸ§¹ ${pages.length} tareas encontradas para archivar`);

    for (const page of pages) {
      await axios.patch(
        `https://api.notion.com/v1/pages/${page.id}`,
        { archived: true },
        {
          headers: {
            Authorization: `Bearer ${NOTION_TOKEN}`,
            'Notion-Version': '2022-06-28',
            'Content-Type': 'application/json'
          }
        }
      );
      console.log(`âœ… PÃ¡gina archivada: ${page.id}`);
    }

    res.send(`ğŸ—‘ï¸ ${pages.length} tareas archivadas correctamente.`);
  } catch (err) {
    console.error('âŒ Error en limpieza:', err.response?.data || err.message);
    res.status(500).send('âš ï¸ Error al limpiar tareas.');
  }
});

// ğŸš€ Levantar el servidor
app.listen(PORT, () => {
  console.log(`ğŸŒ BetsyBot corriendo en http://localhost:${PORT}`);
});
